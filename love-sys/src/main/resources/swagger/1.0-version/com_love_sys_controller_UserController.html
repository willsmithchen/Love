保存用戶信息

作者: lujia chen

请求URL

/user-manager/save-user POST

请求体 application/json

{
	"userId":"long //用户id",
	"deptId":"long //部门id",
	"userName":"string //用户账号",
	"nickName":"string //用户昵称",
	"userType":"string //用户类型",
	"email":"string //用户邮箱",
	"phone":"string //手机号码",
	"sex":"string //用户性别",
	"avatar":"string //头像地址",
	"password":"string //密码",
	"status":"string //账号状态（0正常，1未知）",
	"delFlag":"string //删除标志（0代表存在 2代表删除）",
	"loginIp":"string //最后登录ip",
	"loginDate":"string //最后登录时间",
	"createBy":"string //创建者",
	"createTime":"string //创建时间",
	"updateBy":"string //更新者",
	"updateTime":"string //更新时间",
	"remark":"string //备注",
	"clientId":"string //企业id",
	"userClass":"string //用户类型(1001,云平台 admin 1002,云平台普通用户 1003,企业admin 1004,企业其他账号)"
}

返回结果

{
	"code":"int",
	"message":"string"
}
